 65                         convertedRes = res.payload;

 68                         convertedRes.returnValue = false;

 83     function checkErrorCodeNText(result, errorCode, errorText) {
 84 
 85         if (result.errorCode === undefined || result.errorCode === null ) {
 86             result.errorCode = errorCode;
 87         }
 88         if (result.errorText ===undefined || result.errorText === null) {
 89             result.errorText = errorText;

153                             log("call successCallback");
154                             successCallback();
155                         }
156                     }
157                 },
158                 onFailure: function(result) {
