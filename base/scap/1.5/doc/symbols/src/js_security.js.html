118     Security.prototype.registerServerCertificate = function (successCallback, errorCallback, options) {
119             log("registerServerCertificate: " + JSON.stringify(options));
120 
121             if (options.userName === undefined || typeof options.userName !== 'string' || options.userName.length < 4 || options.userName.length > 10 ||
122                 options.password === undefined || typeof options.password !== 'string' || options.password.length < 4 || options.password.length > 10 ||
123                 options.certificate === undefined || typeof options.certificate !== 'string') {
124 
125                 if (typeof errorCallback === 'function') {
126                     var result = {};
127                     checkErrorCodeNText(result, "SRSC", "Security.registerServerCertificate returns failure. invalid parameters or out of range.");
128                     errorCallback(result);
129                 }
130 
131                 return;
132             }
133 
134             service.Request("luna://com.webos.service.commercial.signage.storageservice/security/", {
135                 method: "registerServerCertificate",
136                 parameters: {
137                     userName : options.userName,
138                     password : options.password,
139                     certificate : options.certificate
140                 },
141                 onSuccess: function(result) {
142                     log("registerServerCertificate: On Success");
143 
144                     if (result.returnValue === true) {
145                         if (typeof successCallback === 'function') {
146                             successCallback();
147                         }
148                     }
149                 },
150                 onFailure: function(result) {
151                     log("registerServerCertificate: On Failure");
152                     delete result.returnValue;
153                     if (typeof errorCallback === 'function') {
154                         checkErrorCodeNText(result, "SRSC", "Security.registerServerCertificate returns failure.");
155                         errorCallback(result);
156                     }
157                 }
158             });
159 
160             log("Security.registerServerCertificate Done");
161     };

276      *   var options = {

282      *      console.log("cbObject : " + JSON.stringify(cbObject));
283      *   }
284      *
285      *   function failureCb(cbObject) {
286      *      var errorCode = cbObject.errorCode;
287      *      var errorText = cbObject.errorText;
288      *      console.log ("Error Code [" + errorCode + "]: " + errorText);
289      *   }

292      *   security.existServerCertificate(successCb, failureCb, options);
293      * }
294      * @since 1.4.1

297             log("existServerCertificate: " + JSON.stringify(options));
298 
299             if (options.userName === undefined || typeof options.userName !== 'string' || options.userName.length < 4 || options.userName.length > 10 ||
300                 options.password === undefined || typeof options.password !== 'string' || options.password.length < 4 || options.password.length > 10) {
301 
302                 if (typeof errorCallback === 'function') {
303                     var result = {};
304                     checkErrorCodeNText(result, "SESC", "Security.existServerCertificate returns failure. invalid parameters or out of range.");
305                     errorCallback(result);
306                 }
307 
308                 return;
309             }
310 
311             service.Request("luna://com.webos.service.commercial.signage.storageservice/security/", {
312                 method: "existServerCertificate",
313                 parameters: {
314                     userName : options.userName,
315                     password : options.password
316                 },
317                 onSuccess: function(result) {
318                     log("existServerCertificate: On Success");
319 
320                     if (result.returnValue === true) {
321                         if (typeof successCallback === 'function') {
322                             var cbObj      = {};
323                             cbObj.userName = options.userName;
324                             cbObj.exist    = result.exist;
325                             successCallback(cbObj);
326                         }
327                     }
328                 },
329                 onFailure: function(result) {
330                     log("existServerCertificate: On Failure");
331                     delete result.returnValue;
332                     if (typeof errorCallback === 'function') {
333                         checkErrorCodeNText(result, "SESC", "Security.existServerCertificate returns failure.");
334                         errorCallback(result);
335                     }
336                 }
337             });
338 
339             log("Security.existServerCertificate Done");
