 54     function checkPlatformVersion(cb) {
 55 
 56         if (version === null) {
 57 
 58             service.Request('luna://com.webos.service.tv.systemproperty', {
 59                 method: 'getSystemInfo',

 68                     if (temp.length >= 1 && temp[0] === '1') {
 69                         platformInfoObj = {
 70                             webOSVer: 1,
 71                             chipset: result.boardType.split("_")[0]
 72                         };
 73                     } else if (temp.length >= 1 && temp[0] === '2') {
 74                         platformInfoObj = {
 75                             webOSVer: 2,
 76                             chipset: result.boardType.split("_")[0]
 77                         };
 78                     } else if (temp.length >= 1 && temp[0] === '3') {
 79                         platformInfoObj = {
 80                             webOSVer: 3,
 81                             chipset: result.boardType.split("_")[0]
 82                         };
 83                     } else {
 84                         platformInfoObj = {
 85                             webOSVer: 0,
 86                             chipset: ""
 87                         };
 88                     }
 89                     version = platformInfoObj.webOSVer;
 90                     cb(platformInfoObj);
 91                 },
 92                 onFailure: function(error) {
 93                     log("getPlatformInfo: onFailure");
 94                     platformInfoObj = {
 95                         webOSVer: 0,
 96                         chipset: ""
 97                     }
 98                     cb(platformInfoObj);
 99                 }
100             });
101 
102         } else {
103             cb(platformInfoObj);
104         }

328         log("addHolidaySchedule: " + "startMonth : " + options.startMonth + ", startDay : " + options.startDay + ", endMonth : " + options.endMonth + ", endDay : " + options.endDay);
329 
330         var ret = { };
331         checkErrorCodeNText(ret, "ERROR", "This function is not supported");
332         errorCallback(ret);
333         return;
334 
335         var startSch = new Date((new Date()).getFullYear(), options.startMonth-1, options.startDay, 0, 0);
336         var endSch = new Date((new Date()).getFullYear(), options.endMonth-1, options.endDay, 0, 0);
337 
338         //validation check
339         if (typeof errorCallback === 'function') {
340             if (typeof options.startMonth !== 'number' || typeof options.startDay !== 'number' || isNaN(options.startMonth) || isNaN(options.startDay) ||
341                     typeof options.endMonth !== 'number' || typeof options.endDay !== 'number' || isNaN(options.endMonth) || isNaN(options.endDay)) {
342                 var result = {};
343                 checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure. parmas are not valid.");
344                 errorCallback(result);
345                 return;
346             }
347 
348             if (startSch.getMonth() !== options.startMonth - 1 || startSch.getDate() !== options.startDay ||
349                 endSch.getMonth() !== options.endMonth - 1 || endSch.getDate() !== options.endDay) {
350                 var result = {};
351                 checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure for out of range.");
352                 errorCallback(result);
353                 return;
354             }
355 
356             if (startSch.getTime() > endSch.getTime())
357             {
358                 var result = {};
359                 checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure. schedule is not valid.");
360                 errorCallback(result);
361                 return;
362             }
363         }
364 
365         service.Request("luna://com.webos.service.commercial.signage.storageservice/settings/", {
366             method : "get",
367             parameters : {
368                 category : "commercial",
369                 keys : ["holidaySchedule"]
370             },
371             onSuccess : function(result) {
372                 log("getHolidaySchedule: On Success");
373 
374                 if (result.returnValue === true) {
375                     if (typeof successCallback === 'function') {
376                         var schedule      = {};
377                         var savedStartSch = null
378                         var savedEndSch   = null
379                         var scheduleList  = result.settings.holidaySchedule;
380 
381                         if(scheduleList.length >= 7)
382                         {
383                             var result = {};
384                             checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure. schedule list is full");
385                             errorCallback(result);
386                             return;
387                         }
388 
389                         //check same schedule
390                         for(var i=0; i<scheduleList.length; i++) {
391 
392                             savedStartSch = new Date((new Date()).getFullYear(), scheduleList[i].startMonth - 1, scheduleList[i].startDay, 0, 0);
393                             savedEndSch   = new Date((new Date()).getFullYear(), scheduleList[i].endMonth - 1, scheduleList[i].endDay, 0, 0);
394 
395                             if ( ((savedStartSch.getTime() <= startSch.getTime()) && (savedEndSch.getTime() > startSch.getTime())) ||
396                                  ((savedStartSch.getTime() < endSch.getTime()) && (savedEndSch.getTime() >= endSch.getTime())) ||
397                                  ((savedStartSch.getTime() === startSch.getTime()) && (savedEndSch.getTime() === endSch.getTime())) ||
398                                  ((startSch.getTime() === endSch.getTime()) && (savedEndSch.getTime() === startSch.getTime())) ) {
399                                 var result = {};
400                                 checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure. schedule is not valid");
401                                 errorCallback(result);
402                                 return;
403                             }
404                         }
405 
406                         schedule.startMonth = options.startMonth;
407                         schedule.startDay   = options.startDay;
408                         schedule.endMonth   = options.endMonth;
409                         schedule.endDay     = options.endDay;
410                         scheduleList.push(schedule);
411 
412                         log("scheduleList : " + JSON.stringify(scheduleList));
413 
414                         service.Request("luna://com.webos.service.commercial.signage.storageservice/settings/", {
415                             method: "set",
416                             parameters: {
417                                 category: "commercial",
418                                 settings: {
419                                     "holidaySchedule": scheduleList
420                                 }
421                             },
422                             onSuccess: function(result) {
423                                 log("addHolidaySchedule: On Success");
424 
425                                 if (result.returnValue === true) {
426                                     if (typeof successCallback === 'function') {
427                                         successCallback();
428                                     }
429                                 }
430                             },
431                             onFailure: function(result) {
432                                 log("addHolidaySchedule: On Failure");
433                                 delete result.returnValue;
434                                 if (typeof errorCallback === 'function') {
435                                     checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure.");
436                                     errorCallback(result);
437                                 }
438                             }
439                         });
440                     }
441                 }
442             },
443             onFailure : function(result) {
444                 log("addHolidaySchedule: On Failure");
445                 delete result.returnValue;
446                 if (typeof errorCallback === 'function') {
447                     checkErrorCodeNText(result, "TAHS", "Time.addHolidaySchedule returns failure.");
448                     errorCallback(result);
