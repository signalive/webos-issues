 31 	function parseUri (str) {
 32 		var	o   = parseUri.options,
 33 		m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
 34 		uri = {},
 35 		i   = 14;
 36 
 37 		while (i--) {uri[o.key[i]] = m[i] || "";}
 38 
 39 		uri[o.q.name] = {};
 40 		uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
 41 			if ($1) {uri[o.q.name][$1] = $2;}
 42 		});
 43 		validateURI(uri);
 44 		return uri;
 45 	}

 47 	parseUri.options = {
 48 			strictMode: false,
 49 			key: ["source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
 50 			q:   {
 51 				name:   "queryKey",
 52 				parser: /(?:^|&)([^&=]*)=?([^&]*)/g
 53 			},
 54 			parser: {
 55 				strict: /^(?:([^:\/?#]+):)?(?:\/\/((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?))?((((?:[^?#\/]*\/)*)([^?#]*))(?:\?([^#]*))?(?:#(.*))?)/,
 56 				loose:  /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*)(?::([^:@]*))?)?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/
 57 			}
 58 	};

 60 	function validateURI(parser){
 61 
 62 		// Take file, http, or https protocol only.
 63 		if(parser.protocol === 'file'){
 64 			if(parser.host === 'internal'){
 65 				return validateInternalURI(parser);
 66 			}
 67 			else if(parser.host === 'usb'){
 68 				validateUsbURI(parser);
 69 			}
 70 			else{
 71 				throw " Invalid Host: " + parser.host;
 72 			}
 73 		}
 74 		else if(parser.protocol === 'http' || parser.protocol === 'https'){
 75 			if(parser.path.charAt(parser.path.length-1) === '/'){
 76 				throw "Invalid path: " + parser.path;
 77 			}
 78 			else{
 79 				return;
 80 			}
 81 		}
 82 		else{
 83 			throw "Invalid protocol: " + parser.protocol;
 84 		}
 85 	}
 86 
 87 	function validateInternalURI(parser){
 88 
 89 		if(parser.path){
 90 			log("Path: " + parser.path);
 91 			if(parser.path.lastIndexOf("/") > 0){
 92 				throw "Invalid internal path: " + parser.path;
 93 			}
 94 			else{
 95 				return;
 96 			}
 97 		}
 98 		else{
 99 			log("No Path: ");
100 			throw "Invalid internal path: " + parser.path;
101 		}
102 	}
103 
104 	function validateUsbURI(parser){
105 		if(parser.port){
106 			if(parser.port.match("/[0-9]+/")){
107 				return {result:true};
108 			}
109 			else{
110 				throw "Invalid USB host: " + parser.host;
111 			}
112 		}
113 		else{
114 			throw "Invalid USB host: " + parser.host;
115 		}
116 
117 		if(parser.path){
118 			if(parser.path.charAt(parser.path.length-1) === '/'){
119 				throw "Invalid USB path: " + parser.path;
120 			}
121 			else{
122 				return {result:true};
123 			}
124 		}
125 		else{
126 			throw "Invalid USB path: " + parser.path;
127 		}
